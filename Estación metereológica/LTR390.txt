#include <Wire.h>
#include <Adafruit_LTR390.h>

Adafruit_LTR390 uv = Adafruit_LTR390();

void setup() {
  Serial.begin(9600);
  if (!uv.begin()) {
    Serial.println("No se encontró el sensor LTR390!");
    while (1);
  }
  Serial.println("Sensor LTR390 encontrado");
}

void loop() {
  uint32_t uv_index = uv.readUVS();
  Serial.print("Índice UV: ");
  Serial.println(uv_index);
  delay(1000);
}
